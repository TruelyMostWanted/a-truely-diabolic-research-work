\begin{itemize}

    \item \textbf{Features:} The set of features to be developed.
    \item \textbf{Users:} The set of users for whom the software is being developed.
    \item \textbf{Sprints:} The set of time-boxed iterations.
    \item \textbf{Stakeholders:} The set of people interested in the software
\end{itemize}

\begin{itemize}
    \item $i$ : Index for teams
    \item $j$ : Index for features
    \item $k$ : Index for sprints
    \item $l$ : Index for stakeholders

    \item \textbf{G0 (Maximize Team Size):} $\max \sum_{i} team\_size_i$
    \item \textbf{G1 (Maximize Sprint Velocity):} $\max \sum_{k} sprint\_velocity_k$
    \item \textbf{G2 (Minimize Blocker Count):} $\min \sum_{j} blocker\_severity_j$
\end{itemize}

\section{Conditions}

\begin{itemize}
    \item \textbf{C0 (Ensure Team Size Within Range):} $2 \le team\_size_i \le 10$

\end{itemize}

\section{Decision Variables}

\begin{itemize}
